{
arr = [ 1, 2, 3, 4, 10, 40, 55 ];
x = 60;

result = 0-1;
l = 0;
r = 6;

while (l <= r and result == 0-1){
    mid = l + (r - l)//2;
    if(arr[mid] == x){
        result = mid;
    }
    else{
        if(arr[mid] < x){
            l = mid + 1;
        }
        else{
            r = mid - 1;
        }
    }

}

print(result);
}
